1. Клонирование репозитория
<!-- Копируем репозиторий с GitHub и переходим в папку проекта -->

bash
git clone https://github.com/rightmelancholy/software_systems_development.git
cd software_systems_development

2. Создание виртуального окружения
<!-- Создаём изолированное Python окружение для избежания конфликтов зависимостей -->

source venv/bin/activate
